#include <stdio.h>
#include <stdlib.h>

void plusMinus(int arr_count, int* arr) {
    int count_positive = 0;
    int count_negative = 0;
    int count_zero = 0;

    for (int i = 0; i < arr_count; i++) {
        if (arr[i] > 0) {
            count_positive++;
        } else if (arr[i] < 0) {
            count_negative++;
        } else {
            count_zero++;
        }
    }

    float total = (float) arr_count;

    printf("%.6f\n", count_positive / total);
    printf("%.6f\n", count_negative / total);
    printf("%.6f\n", count_zero / total);
}

int main() {
    int n;
    scanf("%d", &n);

    int* arr = malloc(n * sizeof(int));

    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    plusMinus(n, arr);

    free(arr);
    return 0;
}

#include <stdio.h>
#include <stdlib.h>

void miniMaxSum(int* arr, int size) {
    long total = 0;
    int min = arr[0];
    int max = arr[0];

    for (int i = 0; i < size; i++) {
        total += arr[i];
        if (arr[i] < min) {
            min = arr[i];
        }
        if (arr[i] > max) {
            max = arr[i];
        }
    }

    long min_sum = total - max;
    long max_sum = total - min;

    printf("%ld %ld\n", min_sum, max_sum);
}

int main() {
    int arr[5];
    for (int i = 0; i < 5; i++) {
        scanf("%d", &arr[i]);
    }

    miniMaxSum(arr, 5);

    return 0;
}

#include <stdio.h>
#include <string.h>

void timeConversion(char* time) {
    int hour;
    char period[3];
    
    sscanf(time, "%2d:%*2d:%*2d%2s", &hour, period);  

    if (strcmp(period, "AM") == 0) {
        if (hour == 12) {
            time[0] = '0';
            time[1] = '0';
        }
    } else {
        if (hour != 12) {
            hour += 12;
            time[0] = '0' + hour / 10;
            time[1] = '0' + hour % 10;
        }
    }

    time[8] = '\0';  // Remove AM or PM
    printf("%s\n", time);
}

int main() {
    char input[11];
    scanf("%s", input);
    timeConversion(input);
    return 0;
}
